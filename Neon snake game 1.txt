<script>
class SnakeGame {
    constructor() {
        this.container = document.getElementById("game-container");
        this.scoreText = document.getElementById("score");
        this.highScoreText = document.getElementById("highScore");

        this.startScreen = document.getElementById("startScreen");
        this.gameOverScreen = document.getElementById("gameOverScreen");
        this.pauseScreen = document.getElementById("pauseScreen");
        this.finalScore = document.getElementById("finalScore");

        this.gridSize = 20;
        this.interval = null;

        this.loadHighScore();
        this.init();
        this.addControls();
    }

    init() {
        this.snake = [{ x: 10, y: 10 }];
        this.direction = { x: 0, y: 0 };
        this.food = {};
        this.score = 0;
        this.speed = 200;
        this.running = false;

        this.spawnFood();
        this.updateUI();
        this.draw();
    }

    start() {
        this.startScreen.classList.add("hidden");
        this.gameOverScreen.classList.add("hidden");

        if (this.interval) clearInterval(this.interval);

        this.running = true;
        this.interval = setInterval(() => this.update(), this.speed);
    }

    restart() {
        clearInterval(this.interval);
        this.gameOverScreen.classList.add("hidden");
        this.init();
        this.start();
    }

    togglePause() {
        if (!this.running) return;

        this.running = false;
        this.pauseScreen.classList.remove("hidden");

        setTimeout(() => {
            this.running = true;
            this.pauseScreen.classList.add("hidden");
        }, 1000);
    }

    update() {
        if (!this.running) return;

        const head = {
            x: this.snake[0].x + this.direction.x,
            y: this.snake[0].y + this.direction.y
        };

        if (head.x < 1 || head.x > this.gridSize || head.y < 1 || head.y > this.gridSize) {
            return this.gameOver();
        }

        if (this.snake.some(p => p.x === head.x && p.y === head.y)) {
            return this.gameOver();
        }

        this.snake.unshift(head);

        if (head.x === this.food.x && head.y === this.food.y) {
            this.score++;
            this.animateScore();
            this.spawnFood();
            this.increaseSpeed();
        } else {
            this.snake.pop();
        }

        this.draw();
    }

    draw() {
        this.container.innerHTML = "";

        this.snake.forEach(part => {
            const el = document.createElement("div");
            el.style.gridColumnStart = part.x;
            el.style.gridRowStart = part.y;
            el.classList.add("snake");
            this.container.appendChild(el);
        });

        const foodEl = document.createElement("div");
        foodEl.style.gridColumnStart = this.food.x;
        foodEl.style.gridRowStart = this.food.y;
        foodEl.classList.add("food");
        this.container.appendChild(foodEl);
    }

    spawnFood() {
        let valid = false;

        while (!valid) {
            this.food = {
                x: Math.floor(Math.random() * this.gridSize) + 1,
                y: Math.floor(Math.random() * this.gridSize) + 1
            };

            valid = !this.snake.some(p => p.x === this.food.x && p.y === this.food.y);
        }
    }

    increaseSpeed() {
        clearInterval(this.interval);
        this.speed *= 0.97;
        this.interval = setInterval(() => this.update(), this.speed);
    }

    gameOver() {
        this.running = false;
        clearInterval(this.interval);

        this.finalScore.innerText = "Score: " + this.score;
        this.gameOverScreen.classList.remove("hidden");

        if (this.score > this.highScore) {
            localStorage.setItem("highScore", this.score);
        }

        this.loadHighScore();
    }

    updateUI() {
        this.scoreText.innerText = "Score: " + this.score;
    }

    animateScore() {
        this.updateUI();
        this.scoreText.style.transform = "scale(1.3)";
        setTimeout(() => {
            this.scoreText.style.transform = "scale(1)";
        }, 100);
    }

    loadHighScore() {
        this.highScore = localStorage.getItem("highScore") || 0;
        this.highScoreText.innerText = "High Score: " + this.highScore;
    }

    setDifficulty(speed) {
        this.speed = parseInt(speed);
        if (this.running) {
            clearInterval(this.interval);
            this.start();
        }
    }

    addControls() {
        document.addEventListener("keydown", (e) => {
            if (e.key === "ArrowUp" && this.direction.y === 0) this.direction = { x: 0, y: -1 };
            if (e.key === "ArrowDown" && this.direction.y === 0) this.direction = { x: 0, y: 1 };
            if (e.key === "ArrowLeft" && this.direction.x === 0) this.direction = { x: -1, y: 0 };
            if (e.key === "ArrowRight" && this.direction.x === 0) this.direction = { x: 1, y: 0 };

            // ENTER = Restart
            if (e.key === "Enter") this.restart();
        });
    }
}

const game = new SnakeGame();
</script>
